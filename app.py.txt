# --- FONCTIONS DE SÉCURITÉ (BLINDAGE ANTI-CRASH) ---
def extraire_json_securise(texte_brut):
    """Extrait chirurgicalement le JSON même si l'IA bavarde autour."""
    try:
        # Cherche tout ce qui ressemble à un objet JSON du premier { au dernier }
        match = re.search(r'\{.*\}', texte_brut, re.DOTALL)
        if match:
            return json.loads(match.group(0))
        else:
            raise ValueError("Aucun bloc JSON détecté dans la réponse.")
    except json.JSONDecodeError as e:
        raise ValueError(f"Le format JSON généré est corrompu : {str(e)}")

def desinfecter_texte_pdf(texte):
    """Supprime tous les caractères spéciaux, accents et émojis pour FPDF."""
    if not texte: return ""
    texte_str = str(texte)
    # Normalise et supprime les accents/émojis pour garantir la compatibilité latin-1
    texte_propre = unicodedata.normalize('NFD', texte_str).encode('ascii', 'ignore').decode('utf-8')
    return texte_propre

# --- FONCTION D'EXPORTATION PDF (SÉCURISÉE) ---
def generer_pdf_robuste(data):
    pdf = FPDF()
    pdf.add_page()
    pdf.set_font("Arial", 'B', 16)
    
    titre = desinfecter_texte_pdf(f"Support Apprenti : {data.get('sujet', 'Sujet inconnu')}")
    pdf.cell(200, 10, txt=titre, ln=True, align='C')
    
    pdf.set_font("Arial", size=12)
    pdf.ln(10)
    
    diplome = desinfecter_texte_pdf(data.get('diplome', ''))
    ref = desinfecter_texte_pdf(data.get('meta', {}).get('referentiel_code', 'Non defini'))
    pdf.multi_cell(0, 10, txt=f"Diplome : {diplome} | Referentiel : {ref}")
    pdf.ln(5)
    
    pdf.set_font("Arial", 'B', 14)
    pdf.cell(200, 10, txt="1. Mission Professionnelle", ln=True)
    pdf.set_font("Arial", size=11)
    
    enonce = desinfecter_texte_pdf(data.get('mission_technique', {}).get('enonce', ''))
    pdf.multi_cell(0, 8, txt=enonce)
    
    # On gère l'erreur d'encodage au moment de la création du fichier
    try:
        return pdf.output(dest='S').encode('latin-1')
    except Exception as e:
        raise ValueError(f"Erreur d'encodage PDF : {str(e)}")

# --- MOTEUR DE GÉNÉRATION V22 ---
def generer_module_indestructible(diplome, sujet, lieu, niveau_bloom):
    moteur = [m.name for m in genai.list_models() if 'generateContent' in m.supported_generation_methods][0]
    # On retire le paramètre mime_type strict qui provoque des crashs internes chez Google
    model = genai.GenerativeModel(moteur)
    
    prompt = f"""
    Agis en expert ingénieur pédagogique du CFA de Chartres. Crée un module pour {diplome} sur le sujet : {sujet}.
    Lieu du scénario : {lieu}. 
    Niveau de difficulté (Taxonomie de Bloom) : {niveau_bloom}. 
    
    REGLE ABSOLUE : Tu ne dois générer AUCUN texte, AUCUN markdown, AUCUN émoji. 
    Tu dois répondre UNIQUEMENT par cet objet JSON brut et valide :
    
    {{
        "meta": {{"referentiel_code": "Ex: C2.1", "temps_estime": "60 min"}},
        "intro": {{"scenario": "Scenario humoristique a Chartres"}},
        "mission_technique": {{"enonce": "Description du defi", "etapes": "Solution detaillee", "complexite": "{niveau_bloom}"}},
        "evaluation": {{"vrai_faux": [{{"text": "Affirmation 1", "val": true, "expl": "Correction"}}]}},
        "autonomie": {{"activite": "Travail de 60 min"}},
        "formateur": {{"critique_pedagogique": "Pourquoi cet exercice est adapte au niveau demande"}}
    }}
    """
    reponse = model.generate_content(prompt)
    return extraire_json_securise(reponse.text)
