# --- FONCTIONS DE NETTOYAGE (ANTI-CRASH) ---
def nettoyer_json(texte_brut):
    """Enlève les balises markdown parasites (```json ...) de l'IA."""
    texte = texte_brut.strip()
    if texte.startswith("```json"):
        texte = texte.replace("```json", "", 1)
    if texte.endswith("```"):
        # Enlève le dernier ``` en partant de la fin
        texte = "".join(texte.rsplit("```", 1))
    return texte.strip()

def nettoyer_pour_pdf(texte):
    """Nettoie les caractères complexes (émojis, etc.) pour éviter le crash FPDF."""
    if not texte: return ""
    # Remplacement des caractères courants qui font planter FPDF
    texte_propre = str(texte).replace('’', "'").replace('œ', 'oe').replace('€', 'Euros')
    # On force l'encodage en latin-1 (ignorera les émojis)
    return texte_propre.encode('latin-1', 'ignore').decode('latin-1')

# --- FONCTION D'EXPORTATION PDF ---
def generer_pdf(data):
    pdf = FPDF()
    pdf.add_page()
    pdf.set_font("Arial", 'B', 16)
    titre_propre = nettoyer_pour_pdf(f"Support Apprenti : {data['sujet']}")
    pdf.cell(200, 10, txt=titre_propre, ln=True, align='C')
    pdf.set_font("Arial", size=12)
    pdf.ln(10)
    
    ref_propre = nettoyer_pour_pdf(f"Diplome : {data['diplome']} | Referentiel : {data.get('meta', {}).get('referentiel_code', '')}")
    pdf.multi_cell(0, 10, txt=ref_propre)
    pdf.ln(5)
    
    pdf.set_font("Arial", 'B', 14)
    pdf.cell(200, 10, txt="1. Mission Professionnelle", ln=True)
    pdf.set_font("Arial", size=11)
    
    enonce_propre = nettoyer_pour_pdf(data.get('mission_technique', {}).get('enonce', ''))
    pdf.multi_cell(0, 8, txt=enonce_propre)
    return pdf.output(dest='S').encode('latin-1')

# --- MOTEUR DE GÉNÉRATION V21.1 ---
def generer_module_v21(diplome, sujet, lieu, niveau_bloom):
    moteur = [m.name for m in genai.list_models() if 'generateContent' in m.supported_generation_methods][0]
    model = genai.GenerativeModel(moteur, generation_config={"response_mime_type": "application/json"})
    
    prompt = f"""
    Agis en expert ingénieur pédagogique. Crée un module pour {diplome} sur {sujet} à {lieu}.
    Niveau de difficulté (Taxonomie de Bloom) : {niveau_bloom}. Ne mets aucun émoji.
    
    STRUCTURE JSON :
    {{
        "meta": {{"referentiel_code": "...", "temps_estime": "60 min"}},
        "intro": {{"scenario": "Scénario local à Chartres sans émojis"}},
        "mission_technique": {{"enonce": "...", "etapes": "...", "complexite": "{niveau_bloom}"}},
        "evaluation": {{"qcm": [], "vrai_faux": []}},
        "autonomie": {{"activite": "..."}},
        "formateur": {{"critique_pedagogique": "..."}}
    }}
    """
    reponse = model.generate_content(prompt)
    texte_nettoye = nettoyer_json(reponse.text)
    return json.loads(texte_nettoye)
